Typical Usage:
(Examples of a typical usage of the component in user code. For more information please see the page Component Code Typical Usage.)

Below are some examples how to use the custom soft characters. 
  LCD1_LoadSoftChar(LCD1_SOFTCHAR_UE, (byte*)LCD1_SoftCharUE);
  LCD1_LoadSoftChar(LCD1_SOFTCHAR_AE, (byte*)LCD1_SoftCharAE);
  LCD1_LoadSoftChar(LCD1_SOFTCHAR_OE, (byte*)LCD1_SoftCharOE);
  LCD1_Write(LCD1_SOFTCHAR_UE); /* ü */
  LCD1_Write(LCD1_SOFTCHAR_AE); /* ä */
  LCD1_Write(LCD1_SOFTCHAR_OE); /* ö */
  LCD1_WriteLn();
  LCD1_WriteString("Z\1rich ae:\2 oe:\3"); /* Zürich ae:ä oe:ü */
Below are some examples how to use the SetEntryMode() method and their outcome. 
static void T1(void) {
  LCD1_Clear(); LCD1_Home();
  LCD1_SetEntryMode(1,0); /* default */
  LCD1_Write('a');
  LCD1_Write('b');
  LCD1_Write('c');
  LCD1_Write('d');
  /* result: "abcd" */
}

static void T2(void) {
  LCD1_Clear(); LCD1_Home();
  LCD1_SetEntryMode(0,1);
  LCD1_Write('a');
  LCD1_Write('b');
  LCD1_Write('c');
  LCD1_Write('d');
  /* result: "    a
               dcb" */
}

static void T3(void) {
  LCD1_Clear(); LCD1_Home();
  LCD1_SetEntryMode(0,0);
  LCD1_Write('a');
  LCD1_Write('b');
  LCD1_Write('c');
  LCD1_Write('d');
  /* result: "a" */
  LCD1_ShiftRight();
  LCD1_ShiftRight();
  LCD1_ShiftRight();
  LCD1_ShiftRight();
  /* result: "    a
               dcb" */
}

static void T4(void) {
  LCD1_Clear(); LCD1_Home();
  LCD1_SetEntryMode(1,1);
  LCD1_Write('a');
  LCD1_Write('b');
  LCD1_Write('c');
  LCD1_Write('d');
  /* result: "" */
  LCD1_ShiftRight();
  LCD1_ShiftRight();
  LCD1_ShiftRight();
  LCD1_ShiftRight();
  /* result: "abcd" */
}

static void T5(void) {
  LCD1_Clear(); LCD1_Home();
  LCD1_GotoXY(1,5);
  LCD1_SetEntryMode(0,0);
  LCD1_Write('a');
  LCD1_Write('b');
  LCD1_Write('c');
  LCD1_Write('d');
  /* result: " dcba" */
}
